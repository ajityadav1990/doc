#!/bin/bash                                                                                                                                                         
#                                                                                                                                                                   
# Script to notify admin user if Linux load crossed certain limit                                                                                                   
# It will send an email notification to admin.                                                                                                                      

# Set up limit below
NOTIFY="10"         

# admin user email id
EMAIL="servers@infobase.in"
EMAIL1="deepak@infobase.in"
# Subject for email        
SUBJECT="Alert $(hostname) load average"

# -----------------------------------------------------------------

# Os Specifc tweaks do not change anything below ;)
OS="$(uname)"                                      
TRUE="1"                                           
if [ "$OS" == "FreeBSD" ]; then                    
        TEMPFILE="$(mktemp /tmp/$(basename $0).tmp.XXX)"
        FTEXT='load averages:'
elif [ "$OS" == "Linux" ]; then
        TEMPFILE="$(mktemp)"
        FTEXT='load average:'
fi

# get first 5 min load
F5M="$(uptime | awk -F "$FTEXT" '{ print $2 }' | cut -d, -f1)"
# 10 min
F10M="$(uptime | awk -F "$FTEXT" '{ print $2 }' | cut -d, -f2)"
# 15 min
F15M="$(uptime | awk -F "$FTEXT" '{ print $2 }' | cut -d, -f3)"

# mail message
# keep it short coz we may send it to page or as an short message (SMS)
echo "Critical Load average Crossed allowed limit $NOTIFY." >> $TEMPFILE
echo "Hostname: $(hostname)" >> $TEMPFILE
echo "Local Date & Time : $(date)" >> $TEMPFILE
echo "Last 5 minutes load : $F5M" >> $TEMPFILE
# Look if it crossed limit
# compare it with last 15 min load average
RESULT=$(echo "$F15M > $NOTIFY" | bc)

# if so send an email
if [ "$RESULT" == "$TRUE" ]; then
        mail -s "$SUBJECT" "$EMAIL" < $TEMPFILE
        mail -s "$SUBJECT" "$EMAIL1" < $TEMPFILE

#Script to count no of html pages in a dir

#!/bin/bash
echo "enter your dir name:"
read mydir
cd /home/$mydir/public_html
pwd
html=`find . -name "*.html" | wc -l`
echo " no of html pages:- $html"
htm=`find . -name "*.htm" | wc -l`
echo " no of htm pages:- $htm"
jsp=`find . -name "*.jsp" | wc -l`
echo " no of jsp pages:- $jsp"
php=`find . -name "*.php" | wc -l`
echo " no of php pages:- $php"
total=`expr $html + $htm + $jsp + $php`
